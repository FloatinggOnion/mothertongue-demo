// Source: https://www.better-auth.com/docs/adapters/drizzle
import { betterAuth } from "better-auth";
import { drizzleAdapter } from "better-auth/adapters/drizzle";
import { db } from "@/db";

export const auth = betterAuth({
  database: drizzleAdapter(db, {
    provider: "sqlite",
  }),
  emailAndPassword: {
    enabled: true,
  },
});

export type Session = typeof auth.$Infer.Session;
```

// Auth schemas
export const SignupSchema = z.object({
  name: z.string().min(1, 'name is required').max(100, 'name must be 100 characters or less'),
  email: z.string().email('invalid email'),
  password: z.string().min(8, 'password must be at least 8 characters').max(255),
});

export const LoginSchema = z.object({
  email: z.string().email('invalid email'),
  password: z.string().min(1, 'password is required'),
});

// Extended EvaluateSchema to include proficiencyLevel (PERS-01)
export const EvaluateSchemaWithLevel = z.object({
  scenarioId: z.string().min(1, 'scenarioId is required'),
  messages: z.array(MessageSchema).min(1, 'messages array must not be empty'),
  proficiencyLevel: ProficiencyLevelSchema,
});

// Type exports
export type SignupRequest = z.infer<typeof SignupSchema>;
export type LoginRequest = z.infer<typeof LoginSchema>;
export type EvaluateRequestWithLevel = z.infer<typeof EvaluateSchemaWithLevel>;
```

Keep the existing EvaluateSchema unchanged for backward compatibility with existing code. The new EvaluateSchemaWithLevel will be used in the evaluate route for session save (Plan 04).

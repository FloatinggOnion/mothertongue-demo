```typescript
import { getScenarioById } from '@/config/scenarios';

interface SessionCardProps {
  scenarioId: string;
  level: 'beginner' | 'intermediate' | 'advanced';
  fluencyScore: number;
  grammarScore: number;
  confidenceScore: number;
  completedAt: Date;
}

export default function SessionCard({
  scenarioId,
  level,
  fluencyScore,
  grammarScore,
  confidenceScore,
  completedAt,
}: SessionCardProps) {
  const scenario = getScenarioById(scenarioId);
  const scenarioName = scenario?.name || 'Unknown Scenario';

  const formatDate = (date: Date) => {
    const d = new Date(date);
    return d.toLocaleDateString('en-US', {
      weekday: 'short',
      year: 'numeric',
      month: 'short',
      day: 'numeric',
    });
  };

  const formatTime = (date: Date) => {
    const d = new Date(date);
    return d.toLocaleTimeString('en-US', {
      hour: 'numeric',
      minute: '2-digit',
    });
  };

  const levelColor = {
    beginner: 'bg-blue-100 text-blue-800',
    intermediate: 'bg-yellow-100 text-yellow-800',
    advanced: 'bg-green-100 text-green-800',
  };

  const scoreColor = (score: number) => {
    if (score >= 4.5) return 'text-green-600 font-bold';
    if (score >= 3.5) return 'text-yellow-600 font-bold';
    return 'text-red-600 font-bold';
  };

  return (
    <div className="bg-white border border-gray-200 rounded-lg p-4 shadow-sm hover:shadow-md transition-shadow">
      <div className="flex items-start justify-between mb-3">
        <div>
          <h3 className="text-lg font-semibold text-gray-900">
            {scenarioName}
          </h3>
          <p className="text-sm text-gray-500">
            {formatDate(completedAt)} at {formatTime(completedAt)}
          </p>
        </div>
        <span className={`text-xs font-semibold px-2.5 py-0.5 rounded-full ${levelColor[level]}`}>
          {level.charAt(0).toUpperCase() + level.slice(1)}
        </span>
      </div>

      <div className="grid grid-cols-3 gap-3">
        <div className="bg-gray-50 p-2 rounded">
          <p className="text-xs text-gray-600 uppercase tracking-wide">
            Fluency
          </p>
          <p className={`text-xl mt-1 ${scoreColor(fluencyScore)}`}>
            {fluencyScore.toFixed(1)}
          </p>
        </div>

        <div className="bg-gray-50 p-2 rounded">
          <p className="text-xs text-gray-600 uppercase tracking-wide">
            Grammar
          </p>
          <p className={`text-xl mt-1 ${scoreColor(grammarScore)}`}>
            {grammarScore.toFixed(1)}
          </p>
        </div>

        <div className="bg-gray-50 p-2 rounded">
          <p className="text-xs text-gray-600 uppercase tracking-wide">
            Confidence
          </p>
          <p className={`text-xl mt-1 ${scoreColor(confidenceScore)}`}>
            {confidenceScore.toFixed(1)}
          </p>
        </div>
      </div>
    </div>
  );
}
```

```typescript
import { auth } from '@/lib/auth';
import { db } from '@/db';
import { drillSessions } from '@/db/schema';
import { headers } from 'next/headers';
import { redirect } from 'next/navigation';
import { eq, desc } from 'drizzle-orm';
import SessionCard from '@/components/SessionCard';

export const metadata = {
  title: 'Session History',
  description: 'View your past drilling sessions and scores',
};

export default async function HistoryPage() {
  // PERS-03: Validate user is authenticated
  const session = await auth.api.getSession({
    headers: await headers(),
  });

  if (!session) {
    redirect('/login');
  }

  let sessions = [];
  let error: string | null = null;

  try {
    // Query all sessions for this user, ordered newest first
    sessions = await db
      .select()
      .from(drillSessions)
      .where(eq(drillSessions.userId, session.user.id))
      .orderBy(desc(drillSessions.completedAt));
  } catch (err) {
    error = 'Failed to load session history';
    console.error('History page query error:', err);
  }

  return (
    <div className="min-h-screen bg-gray-50 py-8 px-4">
      <div className="max-w-6xl mx-auto">
        <h1 className="text-3xl font-bold text-gray-900 mb-2">
          Session History
        </h1>
        <p className="text-gray-600 mb-8">
          View your past drilling sessions and track your progress over time.
        </p>

        {error && (
          <div className="bg-red-100 text-red-800 px-4 py-3 rounded mb-8">
            {error}
          </div>
        )}

        {sessions.length === 0 ? (
          <div className="bg-white border-2 border-dashed border-gray-300 rounded-lg p-8 text-center">
            <p className="text-lg text-gray-600 mb-4">
              No drilling sessions yet.
            </p>
            <p className="text-gray-500">
              Start a drill to begin building your learning history.
            </p>
          </div>
        ) : (
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
            {sessions.map((sess) => (
              <SessionCard
                key={sess.id}
                scenarioId={sess.scenarioId}
                level={sess.level as 'beginner' | 'intermediate' | 'advanced'}
                fluencyScore={sess.fluencyScore}
                grammarScore={sess.grammarScore}
                confidenceScore={sess.confidenceScore}
                completedAt={sess.completedAt}
              />
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
```

This page:
- Is a server component (validates session server-side)
- Redirects to /login if not authenticated (replaces proxy logic as belt-and-suspenders)
- Queries drillSessions for the authenticated user
- Orders by completedAt descending (newest first)
- Renders SessionCard component for each session
- Shows helpful empty state if no sessions exist
- Has responsive grid (1 col mobile, 2 col tablet, 3 col desktop)
- Displays error message if database query fails

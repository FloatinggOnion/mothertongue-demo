```typescript
import { NextRequest, NextResponse } from 'next/server';
import { evaluateConversation } from '@/services/gemini';
import { getScenarioById } from '@/config/scenarios';
import { EvaluateSchema, getZodErrorMessage } from '@/lib/zod-schemas';
import { logError } from '@/lib/logger';
import { auth } from '@/lib/auth';
import { db } from '@/db';
import { drillSessions } from '@/db/schema';

export async function POST(request: NextRequest) {
  let scenarioId: string | undefined;
  let messageCount: number | undefined;

  try {
    const body = await request.json();

    // Validate request body with Zod (now includes proficiencyLevel)
    const validationResult = EvaluateSchema.safeParse(body);
    if (!validationResult.success) {
      return NextResponse.json(
        { error: getZodErrorMessage(validationResult.error) },
        { status: 400 }
      );
    }

    const {
      scenarioId: validatedScenarioId,
      messages,
      proficiencyLevel,
    } = validationResult.data;

    scenarioId = validatedScenarioId;
    messageCount = messages.length;

    const scenario = getScenarioById(scenarioId);
    if (!scenario) {
      return NextResponse.json(
        { error: 'Scenario not found' },
        { status: 404 }
      );
    }

    const evaluation = await evaluateConversation(scenario, messages);

    // PERS-01: Save session summary after successful evaluation
    // Get authenticated user
    const session = await auth.api.getSession({
      headers: request.headers,
    });

    if (session) {
      // User is authenticated — save session summary
      try {
        await db.insert(drillSessions).values({
          id: crypto.randomUUID(),
          userId: session.user.id,
          scenarioId: validatedScenarioId,
          level: proficiencyLevel,
          turnCount: messages.length,
          fluencyScore: evaluation.fluency,
          grammarScore: evaluation.grammar,
          confidenceScore: evaluation.confidence,
          completedAt: new Date(),
        });
      } catch (dbError) {
        // Log database error but don't fail the request
        // User gets their evaluation result even if save fails
        logError('/api/evaluate (session save)', dbError, {
          userId: session.user.id,
          scenarioId: validatedScenarioId,
        });
      }
    }

    return NextResponse.json(evaluation);
  } catch (error) {
    logError('/api/evaluate', error, {
      scenarioId,
      messageCount,
    });
    return NextResponse.json(
      { error: 'Failed to evaluate conversation' },
      { status: 500 }
    );
  }
}
```

import { auth } from '@/lib/auth';
import { db } from '@/db';
import { userProgress } from '@/db/schema';
import { headers } from 'next/headers';
import { redirect } from 'next/navigation';
import { eq } from 'drizzle-orm';

export default async function DrillPage({ params }: { params: { id: string } }) {
  // PERS-02: Restore user's proficiency level from database
  const session = await auth.api.getSession({
    headers: await headers(),
  });

  if (!session) {
    redirect('/login');
  }

  // Get user's current proficiency level
  let proficiencyLevel = 'beginner'; // Default
  try {
    const userProg = await db
      .select()
      .from(userProgress)
      .where(eq(userProgress.userId, session.user.id))
      .limit(1);

    if (userProg.length > 0) {
      proficiencyLevel = userProg[0].proficiencyLevel;
    }
  } catch (error) {
    console.error('Failed to load user proficiency level:', error);
    // Fall through with default level
  }

  // Pass proficiencyLevel to the client component
  return (
    <DrillClient
      scenarioId={params.id}
      initialProficiencyLevel={proficiencyLevel}
    />
  );
}

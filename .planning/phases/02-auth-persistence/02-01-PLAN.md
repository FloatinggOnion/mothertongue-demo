Create src/db/schema.ts with Better Auth generated tables (user, session, account, verification) plus custom application tables (drillSessions, userProgress).

The schema must enforce non-nullable userId on all application tables — this is a critical constraint from ROADMAP.md. Follow the pattern exactly from RESEARCH.md Pattern 5.

File structure:
```typescript
// src/db/schema.ts
import { sqliteTable, text, integer, real } from "drizzle-orm/sqlite-core";

// Better Auth tables (generated via CLI — copy the output here)
export const user = sqliteTable("user", {
  id: text("id").primaryKey(),
  name: text("name"),
  email: text("email").unique(),
  emailVerified: integer("emailVerified", { mode: "boolean" }),
  image: text("image"),
  createdAt: integer("createdAt", { mode: "timestamp" }),
  updatedAt: integer("updatedAt", { mode: "timestamp" }),
});

export const session = sqliteTable("session", {
  id: text("id").primaryKey(),
  expiresAt: integer("expiresAt", { mode: "timestamp" }),
  token: text("token").unique(),
  createdAt: integer("createdAt", { mode: "timestamp" }),
  updatedAt: integer("updatedAt", { mode: "timestamp" }),
  ipAddress: text("ipAddress"),
  userAgent: text("userAgent"),
  userId: text("userId")
    .notNull()
    .references(() => user.id),
});

export const account = sqliteTable("account", {
  id: text("id").primaryKey(),
  accountId: text("accountId"),
  provider: text("provider"),
  providerAccountId: text("providerAccountId"),
  refreshToken: text("refreshToken"),
  accessToken: text("accessToken"),
  expiresAt: integer("expiresAt", { mode: "timestamp" }),
  password: text("password"),
  createdAt: integer("createdAt", { mode: "timestamp" }),
  updatedAt: integer("updatedAt", { mode: "timestamp" }),
  userId: text("userId")
    .notNull()
    .references(() => user.id),
});

export const verification = sqliteTable("verification", {
  id: text("id").primaryKey(),
  identifier: text("identifier"),
  value: text("value"),
  expiresAt: integer("expiresAt", { mode: "timestamp" }),
  createdAt: integer("createdAt", { mode: "timestamp" }),
  updatedAt: integer("updatedAt", { mode: "timestamp" }),
});

// Custom application tables — non-nullable userId on all
export const drillSessions = sqliteTable("drill_sessions", {
  id: text("id").primaryKey(),
  userId: text("user_id")
    .notNull()
    .references(() => user.id),
  scenarioId: text("scenario_id").notNull(),
  level: text("level", { enum: ["beginner", "intermediate", "advanced"] }).notNull(),
  turnCount: integer("turn_count").notNull(),
  fluencyScore: real("fluency_score").notNull(),
  grammarScore: real("grammar_score").notNull(),
  confidenceScore: real("confidence_score").notNull(),
  completedAt: integer("completed_at", { mode: "timestamp" }).notNull(),
});

export const userProgress = sqliteTable("user_progress", {
  userId: text("user_id")
    .primaryKey()
    .references(() => user.id),
  proficiencyLevel: text("proficiency_level", {
    enum: ["beginner", "intermediate", "advanced"],
  })
    .notNull()
    .default("beginner"),
  updatedAt: integer("updated_at", { mode: "timestamp" }).notNull(),
});
```

CRITICAL: Do NOT use optional() on userId fields. Every app row MUST have a non-nullable userId.

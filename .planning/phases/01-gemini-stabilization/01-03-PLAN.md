const fetchEvaluation = async () => {
  if (!scenario || messages.length === 0) return;

  try {
    setEvaluationError(null); // Clear previous error
    setEvaluation(null);

    const response = await fetch('/api/evaluate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        scenarioId: scenario.id,
        messages: messages.filter((m) => m.id !== 'initial'), // Exclude initial message from evaluation
      }),
    });

    if (!response.ok) {
      throw new Error(`Evaluation failed with status ${response.status}`);
    }

    const data = await response.json();
    setEvaluation(data);
    setShowFeedback(true);
  } catch (error) {
    const errorMessage = error instanceof Error ? error.message : 'Failed to evaluate conversation';
    setEvaluationError(errorMessage);
    setShowFeedback(true); // Show FeedbackCard with error state
  }
};
